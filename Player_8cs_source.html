159 private void PlayerMovementInput()
160 {
161 m_XInput = Input.GetAxisRaw( "Horizontal" );
162 m_YInput = Input.GetAxisRaw( "Vertical" );
163
164 // Handle where you press both x and y at the same time.
165 // Move Diagonally with the same speed.
166 if( ( m_XInput != 0 ) || ( m_YInput != 0 ) )
167 {
168 m_XInput = m_XInput * 0.71f;
169 m_YInput = m_YInput * 0.71f;
170 }
171
172 if( ( m_XInput != 0 ) || ( m_YInput != 0 ) )
173 {
174 m_IsRunning = true;
175 m_IsWalking = false;
176 m_IsIdle = false;
177 m_MovementSpeed = Settings.m_RunningSpeed;
178
179 // Capture player direction for save game
180 if( m_XInput < 0 )
181 {
182 m_PlayerDirection = Direction.LEFT;
183 }
184 else if( m_XInput > 0 )
185 {
186 m_PlayerDirection = Direction.RIGHT;
187 }
188 else if( m_YInput < 0 )
189 {
190 m_PlayerDirection = Direction.DOWN;
191 }
192 else
193 {
194 m_PlayerDirection = Direction.UP;
195 }
196 }
197 else if( ( m_XInput == 0 ) && ( m_YInput == 0 ) )
198 {
199 m_IsRunning = false;
200 m_IsWalking = false;
201 m_IsIdle = true;
202 }
203 }
